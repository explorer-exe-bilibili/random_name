cmake_minimum_required(VERSION 3.15)
project(RandomNameTests)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 添加包含目录
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/src)

# 创建测试可执行文件 - test_format_fix
add_executable(test_format_fix test_format_fix.cpp src/core/baseItem/lang.cpp)

# 设置UTF-8编码（Windows）
if(WIN32)
    target_compile_options(test_format_fix PRIVATE /utf-8)
endif()

# 启用测试
enable_testing()
add_test(NAME test_format_fix COMMAND test_format_fix)
